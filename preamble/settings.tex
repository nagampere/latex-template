\makeatletter % "@"記号の有効化

%%% セクション構成の設定 %%%

\newpage
\@startsection{chapter}{1}{\z@}%
{0.0\Cvs \@plus.5\Cdp \@minus.2\Cdp}%
{1.8\Cvs \@plus.3\Cdp}%

}

  (\arabic{subsection})%
}
